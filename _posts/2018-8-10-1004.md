### HDU OJ 1004

em……这个题目比较简单，我觉得考察的是STL的MAP的用法和遍历。

不多赘述。

```C++
#include <iostream>
#include <map>
#include <string>
using namespace std;
using std::map;
using std::string;

int main()
{
	do {
		int n;
		int count = 0;
		cin >> n;
		if (n == 0 || n == EOF)
			break;

		map<string, int> data;
		string result;
		for (int i = 0; i < n; i++)
		{
			string tpstr;
			cin >> tpstr;
			data[tpstr]++;
			//不使用遍历，直接比较
			if (count < data[tpstr])
			{
				count = data[tpstr];
				result = tpstr;
			}
		}

		//使用遍历来完成
		//map<string, int>::iterator it;
		//for (it = data.begin(); it != data.end();it++)
		//{
		//	if (count < it->second)
		//	{
		//		count = it->second;
		//		result = it->first;
		//	}
		//}
		cout << result << endl;
	} while (true);
	return 0;
}
```
